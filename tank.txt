/*
 After playing this game for a while, i realize the biggest challenge is to choose a 'Tech Tree' from a nation to fit in your play style. 
 At Tier VII (Level 7), the playstyle of each 'Tech Tree' starts to become unique and obvious. 
 Therefor, only Tier 7 tanks data are inserted in this database for demonstration purpose.
 Hence, the architect of this database will help better compare the specification of tanks from different nations with different attributes
 and quickly give player an insight of uniqueness of a certain tank.
*/

-- Designed by Tianpeng Gou (12680373)
-- 1st Edition, 2016

DROP TABLE Nation        CASCADE;
DROP TABLE Tank          CASCADE;
DROP TABLE Gun           CASCADE;
DROP TABLE Ammo          CASCADE;
DROP TABLE Fitting       CASCADE;
DROP TABLE Firing        CASCADE;




CREATE TABLE Nation
(
	nationCode  CHAR(2)   NOT NULL, 
	nationName  TEXT   NOT NULL,
	
	CONSTRAINT NationPK PRIMARY KEY (nationCode),
		
		CONSTRAINT di_table_nation_nationcode CHECK (nationCode IN (
			'RU', 'DE', 'US', 'FR', 'GB', 'CN', 'JP', 'CZ'
		)),
		CONSTRAINT di_table_nation_nationName CHECK (nationName IN (
			'Russia', 'Germany', 'USA', 'France', 'United Kingdom', 'China',
			'Japan', 'Czech'
		))
);


CREATE TABLE Tank
(
	nationCode     CHAR(2)   NOT NULL,
	tankName       TEXT   NOT NULL,
	tankType       TEXT   NOT NULL, 
	hitPoint          INTEGER   NOT NULL,  
	maxWeight         FLOAT     NOT NULL, -- tone
	maxEnginePower    INTEGER   NOT NULL, -- hp
	maxSpeed	      INTEGER   NOT NULL, -- km/h
	maxTraverseSpeed  FLOAT     NOT NULL, -- deg/s
	
	CONSTRAINT TankPK PRIMARY KEY (nationCode, tankName),
	CONSTRAINT TankFK FOREIGN KEY (nationCode) 
										REFERENCES Nation ON DELETE RESTRICT,
	
		CONSTRAINT di_table_tank_tankname CHECK (tankName IN (
			'LTTB', 'A-44', 'KV-13', 'T-43', 'IS', 'KV-3',
			'SU-152', 'SU-100M1', 'SU-14-1', 'S-51',
			'SPIC', 'Panther', 'VK3002D', 'Tiger I', 'Tiger P',
			'Sturer Emil', 'Jagdpanther', 'GW Panther', 
			'M41 Bulldog', 'T71', 'T20', 'T29', 'T25/2', 'T25 AT', 'M12', 
			'AMX 13 75', 'AMX M4', 'AMX AC', 'Lorraine 155', 
			'Comet', 'Black Prince', 'Challenger', 'AT 7', 'Crusader SP', 
			'WZ-131', 'T-34-1', 'IS-2', 'Type 5',
			'O-Ni', 
			'T-34/100')
			),
		
		CONSTRAINT di_table_tank_tanktype CHECK (tankType IN (
			'Light', 'Medium', 'Heavy', 'Destroyer', 'SPG'
		)),
		
		CONSTRAINT di_table_tank_hitpoint CHECK (
			(hitPoint >= 340) AND (hitPoint <= 1550)
		),
		
		CONSTRAINT di_table_tank_netWeight_ton CHECK (
			(maxWeight >= 10.0) AND (maxWeight <= 120.0) 
		),
		
		CONSTRAINT di_table_tank_maxEnginePower_hp CHECK (
			(maxEnginePower >= 100) AND (maxEnginePower <= 1300) 
		),
		
		CONSTRAINT di_table_tank_maxSpeed_kmh CHECK (
			(maxSpeed >= 10) AND (maxSpeed <= 100) 
		),
		
		CONSTRAINT di_table_tank_maxTraverseSpeed_degs CHECK (
			(maxTraverseSpeed >= 10.0) AND (maxTraverseSpeed <= 60.0) 
		)
		
);
 

CREATE TABLE Gun	
(
	gunNameCaliber    TEXT   NOT NULL,
	caliber      INTEGER  NOT NULL, -- mm
	weight       FLOAT    NOT NULL, -- tone
	accuracy     FLOAT    NOT NULL, -- ratio
	aimTime      FLOAT    NOT NULL, -- s
	gunType      TEXT     NOT NULL, 
	
	CONSTRAINT GunPK PRIMARY KEY (gunNameCaliber),
	
	
		CONSTRAINT di_table_gun_caliber CHECK (
			(caliber >= 50) AND (caliber <= 210) 
		),
		
		CONSTRAINT di_table_gun_weight CHECK (
			(weight >= 0.0) AND (weight <= 10.0) 
		),
		
		CONSTRAINT di_table_gun_accuracy CHECK (
			(accuracy >= 0.25) AND (accuracy <= 0.90) 
		),
		
		CONSTRAINT di_table_gun_aimtime CHECK (
			(aimTime >= 0.0) AND (aimTime <= 10.0)  
		),
		CONSTRAINT di_table_gun_guntype CHECK ( gunType IN (
			'manual', 'auto'
		))
);


CREATE TABLE Ammo
(
	ammoName   TEXT   NOT NULL,
	ammoType   TEXT   NOT NULL,
	
	CONSTRAINT AmmoPK PRIMARY KEY (ammoName),
	
	
		CONSTRAINT di_table_ammo_ammoType CHECK (ammoType IN (
			'AP', 'APCR', 'HEAT', 'HE'
		))
	
);


CREATE TABLE Fitting
(
	nationCode        CHAR(2)   NOT NULL,
	tankName          TEXT   NOT NULL,
	gunNameCaliber    TEXT   NOT NULL, -- mm
	reloadTime        FLOAT  NOT NULL, -- s
	depressionAngle   FLOAT  NOT NULL, -- deg
	elevationAngle    FLOAT  NOT NULL, -- deg
	
	CONSTRAINT FittingPK PRIMARY KEY (nationCode, tankName, gunNameCaliber),
	
	CONSTRAINT Fitting_TankFK FOREIGN KEY (nationCode, tankName)
														REFERENCES Tank ON DELETE CASCADE,
	
	CONSTRAINT Fitting_GunFK FOREIGN KEY (gunNameCaliber)
														REFERENCES Gun ON DELETE CASCADE, 
	
		CONSTRAINT di_table_fitting_reloadtime CHECK (
			(reloadTime >= 0.0) AND (reloadTime <= 60) -- Modify later
		),
		
		CONSTRAINT di_table_fitting_depressionangle CHECK (
			(depressionAngle >= -20.0) AND (depressionAngle <= 0.0) -- Modify later
		),
		
		CONSTRAINT di_table_fitting_elevationangle CHECK (
			(elevationAngle >= 0.0) AND (elevationAngle <= 60.0) -- Modify later
		)
);


CREATE TABLE Firing
(
	gunNameCaliber  TEXT      NOT NULL, 
	ammoName        TEXT      NOT NULL,
	penetration     INTEGER   NOT NULL, -- mm
	damage          INTEGER   NOT NULL, 
	
	CONSTRAINT FiringPK PRIMARY KEY (gunNameCaliber, ammoName),
	
	CONSTRAINT Firing_GunFK FOREIGN KEY (gunNameCaliber) 
												REFERENCES Gun ON DELETE CASCADE,
	
	CONSTRAINT Firing_AmmoFK FOREIGN KEY (ammoName)
												REFERENCES Ammo ON DELETE CASCADE, 
	
		CONSTRAINT di_table_firing_penetration CHECK (
			(penetration >= 0) AND (penetration <= 300)  --Modify later
		),
		
		CONSTRAINT di_table_firing_damage CHECK (
			(damage >= 0) AND (damage <= 2000) --Modify later
		)
);
